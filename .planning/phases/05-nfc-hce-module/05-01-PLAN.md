    ```kotlin
    object NfcPayloadBuilder {
        const val VERSION: Byte = 0x01
        const val FLAGS_ENCRYPTED: Byte = 0x00
        const val PAYLOAD_SIZE = 36
        const val PUBKEY_SIZE = 32

        fun buildNfcPayload(pubKeyBytes: ByteArray): ByteArray {
            require(pubKeyBytes.size == PUBKEY_SIZE) { "Ed25519 pubkey must be exactly 32 bytes" }
            val buf = ByteArray(PAYLOAD_SIZE)
            buf[0] = VERSION
            buf[1] = FLAGS_ENCRYPTED
            pubKeyBytes.copyInto(buf, 2)
            val crc = crc16Ccitt(buf, 0, 34)
            buf[34] = (crc shr 8).toByte()
            buf[35] = crc.toByte()
            return buf
        }

        fun validateNfcPayload(payload: ByteArray): Boolean {
            if (payload.size != PAYLOAD_SIZE) return false
            val expected = crc16Ccitt(payload, 0, 34)
            val actual = ((payload[34].toInt() and 0xFF) shl 8) or (payload[35].toInt() and 0xFF)
            return expected == actual
        }

        fun extractPubKey(payload: ByteArray): ByteArray? {
            if (!validateNfcPayload(payload)) return null
            return payload.copyOfRange(2, 34)
        }

        fun crc16Ccitt(data: ByteArray, offset: Int = 0, length: Int = data.size): Int {
            var crc = 0xFFFF
            for (i in offset until offset + length) {
                crc = crc xor ((data[i].toInt() and 0xFF) shl 8)
                repeat(8) { crc = if (crc and 0x8000 != 0) (crc shl 1) xor 0x1021 else crc shl 1 }
                crc = crc and 0xFFFF
            }
            return crc
        }
    }
    ```

    Using Kotlin-inline CRC-16 per Claude's discretion (research recommends this over FFI for trivial computation). CRC-16/CCITT-FALSE: poly=0x1021, init=0xFFFF, no reflection.

    ```kotlin
    class PktapHceService : HostApduService() {
        companion object {
            @Volatile var cachedPayload: ByteArray? = null

            val SW_OK = byteArrayOf(0x90.toByte(), 0x00)
            val SW_UNKNOWN = byteArrayOf(0x6F.toByte(), 0x00)
            val SW_FILE_NOT_FOUND = byteArrayOf(0x6A.toByte(), 0x82.toByte())

            const val EXCHANGE_CLA: Byte = 0x90.toByte()  // Proprietary class
            const val EXCHANGE_INS: Byte = 0x01            // EXCHANGE_KEY instruction
            private const val AID_INS_SELECT: Byte = 0xA4.toByte()
        }

        private var selectAidReceived = false

        override fun processCommandApdu(commandApdu: ByteArray, extras: Bundle?): ByteArray {
            // CRITICAL: Blanket try/catch — unhandled exception permanently kills HCE (Pitfall 1)
            return try {
                val result = PktapApduProtocol.handleApdu(commandApdu, cachedPayload, selectAidReceived)
                selectAidReceived = result.newSelectAidState
                result.peerPayload?.let { NfcExchangeFlow.peerKeyFlow.tryEmit(it) }
                result.responseBytes
            } catch (e: Exception) {
                SW_UNKNOWN
            }
        }

        override fun onDeactivated(reason: Int) {
            selectAidReceived = false
        }
    }
    ```

    `PktapApduProtocol` is an `object` with pure function `handleApdu` implementing:
    - If INS == 0xA4 (SELECT): return SW_OK, set selectAidReceived = true
    - If CLA == 0x90 && INS == 0x01 && selectAidReceived: extract 36 bytes from data field (offset 5..41), return cachedPayload + SW_OK. If cachedPayload null, return SW_FILE_NOT_FOUND.
    - Else: return SW_UNKNOWN

    The EXCHANGE APDU layout (reader side sends): CLA=0x90, INS=0x01, P1=0x00, P2=0x00, LC=0x24 (36), DATA=36 bytes. Total command = 41 bytes minimum.

    ZERO calls to PktapBridge, no crypto, no I/O in processCommandApdu per NFC-03.

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| NFC RF link -> processCommandApdu | Untrusted APDU bytes from any NFC device (not necessarily PKTap) |
| PktapHceService -> NfcExchangeFlow | Peer payload crosses from service to app ViewModel |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-05-01 | Tampering | processCommandApdu | mitigate | Validate APDU size >= 41 before extracting data; copy exactly 36 bytes by offset (not variable length); CRC-16 validation on extracted payload before trusting it in Plan 02 |
| T-05-02 | Denial of Service | processCommandApdu | mitigate | Blanket try/catch wrapping entire method body; return SW_UNKNOWN on any exception; never let exception escape (Pitfall 1 — permanent HCE death) |
| T-05-03 | Spoofing | NFC peer key | accept | NFC layer is key transport only; cryptographic verification (Ed25519 signature) happens in post-tap Rust FFI (decrypt_and_verify in Phase 1); spoofed key will fail signature verification |
| T-05-04 | Information Disclosure | cachedPayload | accept | Payload contains only the Ed25519 public key (not secret) plus version/flags/CRC; no sensitive data exposed over NFC |
| T-05-05 | Denial of Service | SharedFlow overflow | accept | DROP_OLDEST with extraBufferCapacity=1; worst case is a missed tap requiring user retry; no data loss or security impact |
</threat_model>
